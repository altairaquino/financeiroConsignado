CREATE PROCEDURE SP_REGISTRA_ESTORNO_CONTRATO(
  CTNCODG INTEGER,
  DATA DATE)
AS
DECLARE VARIABLE STATUS INTEGER;
DECLARE VARIABLE DATA_CADASTRO DATE;
BEGIN
  SELECT CTNCGSC, CTDCADT FROM CONTRATO WHERE CTNCODG = :CTNCODG
  INTO :STATUS, :DATA_CADASTRO;
  
  IF (STATUS IN (6) AND DATA > DATA_CADASTRO) THEN
  BEGIN
       UPDATE CONTRATO SET CTDEXTN = :DATA, CTNCGSC = 4
       WHERE CTNCODG = :CTNCODG;
  END
  
END^

SET TERM ; ^
