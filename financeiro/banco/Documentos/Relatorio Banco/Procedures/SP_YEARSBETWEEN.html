CREATE PROCEDURE SP_YEARSBETWEEN(
  CAL1 DATE,
  CAL2 DATE)
RETURNS(
  IDADE INTEGER)
AS
DECLARE VARIABLE CAL3 DATE;
DECLARE VARIABLE YEAR1 INTEGER;
DECLARE VARIABLE YEAR2 INTEGER;
DECLARE VARIABLE MONTH1 INTEGER;
DECLARE VARIABLE MONTH2 INTEGER;
DECLARE VARIABLE DAY1 INTEGER;
DECLARE VARIABLE DAY2 INTEGER;
BEGIN
    IF(CAL1 > CAL2)THEN
    BEGIN
         CAL3 = CAL2;
         CAL2 = CAL1;
         CAL1 = CAL3;
    END
	YEAR1 = EXTRACT(YEAR FROM CAL1);
	YEAR2 = EXTRACT(YEAR FROM CAL2);
	MONTH1 = EXTRACT(MONTH FROM CAL1);
	MONTH2 = EXTRACT(MONTH FROM CAL2);
	DAY1 = EXTRACT(DAY FROM CAL1);
	DAY2 = EXTRACT(DAY FROM CAL2);
	IDADE = YEAR2 - YEAR1;
	IF((MONTH2 < MONTH1) OR ((MONTH2 = MONTH1) AND (DAY2 < DAY1)))THEN
       IDADE = IDADE - 1;
    SUSPEND;
END^

SET TERM ; ^
