CREATE PROCEDURE SP_CRIA_TABELAFUNCIONARIO(
  FCANCODG INTEGER)
AS
DECLARE VARIABLE TABELA INTEGER;
DECLARE VARIABLE TIPO INTEGER;
DECLARE VARIABLE PRAZO INTEGER;
DECLARE VARIABLE ZERA CHAR(1);
DECLARE VARIABLE CMSNPERC FLOAT;
DECLARE VARIABLE CMSNPC100 FLOAT;
DECLARE VARIABLE CMSNPC150 FLOAT;
DECLARE VARIABLE PERC_COMIS FLOAT;
BEGIN

  SELECT FCANCGTFA FROM FUNCIONARIOAGENCIA
  WHERE FCANCODG = :FCANCODG
  INTO :TIPO;

  FOR SELECT TPNCODG, TPLZERA, TPNPRAZ, CMSNPERC, CMSNPC100, CMSNPC150
  FROM COMISSAO_SINERGIA
  INNER JOIN TABELAPRODUTO ON
        TPNCODG = CMSNCGTP
  WHERE TPLATIV = 'T'
  AND CMSNCGTFC = :TIPO
  INTO :TABELA, :ZERA, :PRAZO, :CMSNPERC, :CMSNPC100, :CMSNPC150
  DO BEGIN
     INSERT INTO TABELAFUNCIONARIO (TFCNCGFCA, TFCNCGTP, TFCNCOMS, TFCNCM70, TFCNCM100)
     VALUES (:FCANCODG, :TABELA, :CMSNPERC, :CMSNPC100, :CMSNPC150);
  END

END^

SET TERM ; ^
