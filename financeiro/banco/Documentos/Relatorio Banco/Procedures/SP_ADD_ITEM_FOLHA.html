CREATE PROCEDURE SP_ADD_ITEM_FOLHA(
  FOENCGFOL INTEGER)
AS
DECLARE VARIABLE FOENCODG INTEGER;
DECLARE VARIABLE EMPYBASE FLOAT;
DECLARE VARIABLE EMPDADMS DATE;
DECLARE VARIABLE FOLDMES DATE;
DECLARE VARIABLE FOLNDIAS INTEGER;
DECLARE VARIABLE DIAS INTEGER;
DECLARE VARIABLE LIMINSS FLOAT;
DECLARE VARIABLE INSS FLOAT;
DECLARE VARIABLE FGTS FLOAT;
DECLARE VARIABLE SFAM FLOAT;
DECLARE VARIABLE FOENCGEMP INTEGER;
DECLARE VARIABLE QTD INTEGER;
DECLARE VARIABLE DDEP FLOAT;
DECLARE VARIABLE IRRF FLOAT;
BEGIN

DELETE FROM FOLHA_EMP_ITEM WHERE
EXISTS(
SELECT FEINCGFOE FROM FOLHA_EMP WHERE
FEINCGFOE = FOENCODG AND
FOENCGFOL = :FOENCGFOL
);

FOR
SELECT
   FOENCODG,
   EMPYBASE,
   EMPDADMS,
   FOLDMES,
   FOLNDIAS
 FROM
   FOLHA_EMP INNER JOIN EMPREGADO ON
   FOENCGEMP = EMPNCODG
   LEFT JOIN FOLHA ON
   FOLNCODG = FOENCGFOL
 WHERE
   FOENCGFOL = :FOENCGFOL AND
   ( EMPLATIV = 'T' AND  EMPDDEMS IS NULL)
 INTO
   :FOENCODG,
   :EMPYBASE,
   :EMPDADMS,
   :FOLDMES,
   :FOLNDIAS
 DO
BEGIN
/*DERNCODG = 1  - Salário Base*/
IF(EMPDADMS IS NULL)THEN
DIAS = 0;
ELSE BEGIN
IF(FOLDMES < EMPDADMS)THEN

/*DIAS = udf_DaysBetween(EMPDADMS,addDay(FOLDMES,FOLNDIAS));*/
SELECT DAYS FROM
SP_DAYSBETWEEN(:EMPDADMS,ADDDAY(:FOLDMES,:FOLNDIAS))
INTO :DIAS;
IF(DIAS > FOLNDIAS)THEN
DIAS = 0;
ELSE
DIAS = FOLNDIAS;
END
IF(DIAS > 0)THEN
INSERT INTO
  FOLHA_EMP_ITEM(
  FEINCGFOE,
  FEINCGDER,
  FEIYVALOR,
  FEIYBASE)
VALUES(
  :FOENCODG,
  1,
  :EMPYBASE * (:DIAS/CAST(:FOLNDIAS AS FLOAT)),
  :DIAS);
END
/*DERNCODG = 11  - INSS*/
FOR
 SELECT
   FEINCGFOE,
   FEIYVALOR
 FROM
   FOLHA_EMP INNER JOIN FOLHA_EMP_ITEM ON
   FOENCODG = FEINCGFOE
 WHERE
   FEINCGDER = 1 AND
   FOENCGFOL = :FOENCGFOL
 INTO
 :FOENCODG,
 :EMPYBASE
DO BEGIN
SELECT
 MAX(TABNLIMT)
FROM
   TABELA
WHERE
   TABNCGTTB = 1
INTO
   :LIMINSS;
IF(EMPYBASE < LIMINSS)THEN
LIMINSS = EMPYBASE;
SELECT FIRST 1
   TABNPDES
 FROM
   TABELA
 WHERE
   TABNCGTTB = 1 AND
   (TABNLIMT >= :LIMINSS)
 ORDER BY TABNLIMT
 INTO
  :INSS;

INSERT INTO
  FOLHA_EMP_ITEM(
  FEINCGFOE,
  FEINCGDER,
  FEIYVALOR,
  FEIYBASE)
VALUES(
  :FOENCODG,
  11,
  (:LIMINSS * :INSS) / 100,
  :EMPYBASE);
END


/*DERNCODG = 13  - FGTS*/

SELECT FIRST 1 PARNFGTS FROM PARAMETRO INTO :FGTS;
FOR
 SELECT
   FEINCGFOE,
   FEIYVALOR
 FROM
   FOLHA_EMP INNER JOIN FOLHA_EMP_ITEM ON
   FOENCODG = FEINCGFOE
 WHERE
   FEINCGDER = 1 AND
   FOENCGFOL = :FOENCGFOL
 INTO
 :FOENCODG,
 :EMPYBASE
DO BEGIN

INSERT INTO
  FOLHA_EMP_ITEM(
  FEINCGFOE,
  FEINCGDER,
  FEIYVALOR,
  FEIYBASE)
VALUES(
  :FOENCODG,
  13,
  (:EMPYBASE * :FGTS) / 100,
  :EMPYBASE);
END
/*DERNCODG = 12  - Salário Família*/
FOR
 SELECT
   FEINCGFOE,
   FEIYVALOR,
   FOENCGEMP
 FROM
   FOLHA_EMP INNER JOIN FOLHA_EMP_ITEM ON
   FOENCODG = FEINCGFOE
 WHERE
   FEINCGDER = 1 AND
   FOENCGFOL = :FOENCGFOL
 INTO
 :FOENCODG,
 :EMPYBASE,
 :FOENCGEMP
DO BEGIN

 SELECT FIRST 1
   IIF(TABNLIMT >=:EMPYBASE,TABNPARC,NULL)
 FROM
   TABELA
 WHERE
   TABNCGTTB = 3
ORDER BY TABNLIMT
INTO :SFAM;

 SELECT
 COUNT(DEPNCODG)
 FROM
   DEPENDENTE LEFT JOIN EMPREGADO ON
   DEPNCGEN = EMPNCGCEN
 WHERE
   EXISTS(
   SELECT PARNCODG FROM PARAMETRO WHERE
   /*PARNMFAM >= DEPNIDAD*/
   PARNMFAM >= (SELECT IDADE FROM SP_YEARSBETWEEN('NOW',DEPDNASC))
   ) AND
   EMPNCODG =:FOENCGEMP
 INTO :QTD;

IF(QTD > 0 AND SFAM IS NOT NULL)THEN
INSERT INTO
  FOLHA_EMP_ITEM(
  FEINCGFOE,
  FEINCGDER,
  FEIYVALOR,
  FEIYBASE)
VALUES(
  :FOENCODG,
  12,
  :SFAM * :QTD,
  :QTD);
END


/*DERNCODG = 62  - Dependentes de Imposto Renda */
 SELECT FIRST 1
   PARYDDEP
 FROM
   PARAMETRO
INTO :DDEP;
FOR
 SELECT
   FEINCGFOE,
   FEIYVALOR,
   FOENCGEMP
 FROM
   FOLHA_EMP INNER JOIN FOLHA_EMP_ITEM ON
   FOENCODG = FEINCGFOE
 WHERE
   FEINCGDER = 1 AND
   FOENCGFOL = :FOENCGFOL
 INTO
 :FOENCODG,
 :EMPYBASE,
 :FOENCGEMP
DO BEGIN

 SELECT
 COUNT(DEPNCODG)
 FROM
   DEPENDENTE LEFT JOIN EMPREGADO ON
   DEPNCGEN = EMPNCGCEN
 WHERE
   EMPNCODG =:FOENCGEMP
 INTO :QTD;

IF(QTD > 0)THEN
INSERT INTO
  FOLHA_EMP_ITEM(
  FEINCGFOE,
  FEINCGDER,
  FEIYVALOR,
  FEIYBASE)
VALUES(
  :FOENCODG,
  62,
  :DDEP * :QTD,
  :QTD);
END


/*DERNCODG = 10  - Imposto de Renda*/

FOR
SELECT FEINCGFOE,
       FEIYVALOR,
       CAST((FEIYVALOR * TABNPDES)/100 -  TABNPARC AS NUMERIC(13,2)) VALOR
FROM (
      SELECT FEINCGFOE,
             FEIYVALOR,
             MIN(TABNLIMT) MIN_
      FROM (
      SELECT FEINCGFOE FEINCGFOE,
            SUM(IIF(IDRCTIPO = '+',FEIYVALOR,IIF(IDRCTIPO = '-',-FEIYVALOR,NULL)))FEIYVALOR
      FROM FOLHA_EMP_ITEM
           INNER JOIN ITEM_DESC_REND ON IDRNCGDER = FEINCGDER
      WHERE IDRNCGTDR = 1 /*IMPOSTO DE RENDA*/
      GROUP BY FEINCGFOE
           ) IRRF
           LEFT JOIN TABELA ON TABNCGTTB = 2 AND TABNLIMT >= FEIYVALOR
      GROUP BY FEINCGFOE,
               FEIYVALOR
     )
     INNER JOIN TABELA ON TABNCGTTB = 2 AND TABNLIMT = MIN_ AND TABNPDES > 0
     INNER JOIN FOLHA_EMP ON
     FEINCGFOE = FOENCODG
 WHERE
   FOENCGFOL = :FOENCGFOL
 INTO
 :FOENCODG,
 :EMPYBASE,
 :IRRF
DO BEGIN

INSERT INTO
  FOLHA_EMP_ITEM(
  FEINCGFOE,
  FEINCGDER,
  FEIYVALOR,
  FEIYBASE)
VALUES(
  :FOENCODG,
  10,
  :IRRF,
  :EMPYBASE);
END

UPDATE FOLHA_EMP
SET FOEYLIQD = COALESCE((
                         SELECT SUM(IIF(DERCOPER = '+', FEIYVALOR, IIF(DERCOPER
                         = '-', - FEIYVALOR, NULL))) FEIYVALOR
                         FROM FOLHA_EMP_ITEM
                              INNER JOIN DESC_REND ON FEINCGDER = DERNCODG
                         WHERE DERCOPER IN ('+', '-') AND
                               FEINCGFOE = FOENCODG
                         GROUP BY FEINCGFOE
    ), 0),

FOENQTCC =(
SELECT COUNT(DISTINCT ECCNCGCR)
FROM EMPREGADO_CENTROCUSTO
WHERE ECCNCGEMP = FOENCGEMP
GROUP BY ECCNCGEMP)
WHERE FOENCGFOL = :FOENCGFOL;
/*
 SELECT
   count(distinct ECCNCGCR)
 FROM
   EMPREGADO_CENTROCUSTO
 where
   ECCNCGEMP = :ECCNCGEMP
 group by ECCNCGEMP
*/

END^

SET TERM ; ^
