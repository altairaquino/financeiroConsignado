CREATE PROCEDURE SP_INSERE_CONTRATO_COMPLETO(
  ENCNOME VARCHAR(60) CHARACTER SET NONE,
  ENCDOCM VARCHAR(11) CHARACTER SET NONE,
  ENDNASC DATE,
  ENCSEXO CHAR(1) CHARACTER SET NONE,
  ENCRG VARCHAR(14) CHARACTER SET NONE,
  ENDEXRG DATE,
  ENCOERG VARCHAR(5) CHARACTER SET NONE,
  ENCUFRG VARCHAR(2) CHARACTER SET NONE,
  ENCDORG VARCHAR(200) CHARACTER SET NONE,
  ENCFONE VARCHAR(10) CHARACTER SET NONE,
  ENNCGTL INTEGER,
  ENLENDR VARCHAR(80) CHARACTER SET NONE,
  ENNCGCD INTEGER,
  ENCCEP VARCHAR(8) CHARACTER SET NONE,
  ENCBAIR VARCHAR(50) CHARACTER SET NONE,
  CTNNUMR VARCHAR(15) CHARACTER SET NONE,
  CTNCGEN INTEGER,
  CTNCADT INTEGER,
  CTNCGFP INTEGER,
  CTNCGTP INTEGER,
  CTYVALR FLOAT,
  CTNCGBC INTEGER,
  CTCAGEN VARCHAR(6) CHARACTER SET NONE)
AS
DECLARE VARIABLE ENNCODG INTEGER;
DECLARE VARIABLE CTNCODG INTEGER;
BEGIN

     IF (EXISTS(SELECT FIRST 1 ENNCODG FROM ENTIDADE WHERE ENCDOCM = :ENCDOCM AND ENNCGTE = 1)) THEN
     BEGIN
          SELECT FIRST 1 ENNCODG FROM ENTIDADE WHERE ENCDOCM = :ENCDOCM AND ENNCGTE = 1
          INTO :ENNCODG;

          UPDATE ENTIDADE SET ENCNOME = :ENCNOME, ENCSEXO = :ENCSEXO,
          ENDNASC = :ENDNASC, ENDEXRG = :ENDEXRG, ENCRG = :ENCRG, ENCOERG = :ENCOERG,
          ENCUFRG = :ENCUFRG, ENCDORG = :ENCDORG, ENCFONE = :ENCFONE, ENNCGTL = :ENNCGTL,
          ENLENDR = :ENLENDR, ENNCGCD = :ENNCGCD, ENCCEP = :ENCCEP, ENCBAIR = :ENCBAIR
          WHERE ENNCODG = :ENNCODG;

          INSERT
          INTO CONTRATO(CTNNUMR, CTNC2EN, CTNCGEN, CTYVALR, CTNCGFP, CTNCGTP,
          CTNCADT, CTNCGBC, CTCAGEN)
          VALUES (:CTNNUMR, :ENNCODG, :CTNCGEN, :CTYVALR, :CTNCGFP, :CTNCGTP,
          :CTNCADT, :CTNCGBC, :CTCAGEN);
     
     END
     ELSE
     BEGIN
          ENNCODG  =  GEN_ID(ENNCODG_GEN,1);

          INSERT
          INTO ENTIDADE(ENNCODG, ENCNOME, ENNCGTE, ENNCGPP, ENCSEXO, ENCDOCM,
          ENDNASC, ENDEXRG, ENCRG, ENCOERG, ENCUFRG, ENCDORG, ENCFONE, ENNCGTL,
          ENLENDR, ENNCGCD, ENCCEP, ENCBAIR)
          VALUES (:ENNCODG, :ENCNOME, 1, 1, :ENCSEXO, :ENCDOCM, :ENDNASC, :ENDEXRG,
          :ENCRG, :ENCOERG, :ENCUFRG, :ENCDORG, :ENCFONE, :ENNCGTL, :ENLENDR,
          :ENNCGCD, :ENCCEP, :ENCBAIR);
          
          INSERT
          INTO CONTRATO(CTNNUMR, CTNC2EN, CTNCGEN, CTYVALR, CTNCGFP, CTNCGTP,
          CTNCADT, CTNCGBC, CTCAGEN)
          VALUES (:CTNNUMR, :ENNCODG, :CTNCGEN, :CTYVALR, :CTNCGFP, :CTNCGTP,
          :CTNCADT, :CTNCGBC, :CTCAGEN);
     
     END
     
     

     

END^

SET TERM ; ^
