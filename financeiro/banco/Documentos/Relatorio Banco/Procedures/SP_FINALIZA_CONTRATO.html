CREATE PROCEDURE SP_FINALIZA_CONTRATO(
  CTNCODG INTEGER)
AS
DECLARE VARIABLE STATUS INTEGER;
BEGIN
  SELECT CTNCGSC FROM CONTRATO WHERE CTNCODG = :CTNCODG
  INTO :STATUS;

  IF (STATUS NOT IN (4)) THEN
  BEGIN
       UPDATE CONTRATO SET CTNCGSC = 6
       WHERE CTNCODG = :CTNCODG;
  END

END^

SET TERM ; ^
