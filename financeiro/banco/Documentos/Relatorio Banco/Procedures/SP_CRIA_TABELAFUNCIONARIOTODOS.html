CREATE PROCEDURE SP_CRIA_TABELAFUNCIONARIOTODOS(
  TPNCODG INTEGER)
AS
DECLARE VARIABLE PERC_COMIS FLOAT;
DECLARE VARIABLE PRAZO INTEGER;
DECLARE VARIABLE FCANCODG INTEGER;
DECLARE VARIABLE ZERA CHAR(1);
DECLARE VARIABLE TIPO INTEGER;
BEGIN

  SELECT TPLZERA, TPNPRAZ FROM TABELAPRODUTO
  WHERE TPNCODG = :TPNCODG
  INTO :ZERA, :PRAZO;

  FOR SELECT FCANCODG, FCANPERC FROM FUNCIONARIOAGENCIA
  INTO :FCANCODG, :PERC_COMIS
  DO BEGIN
     INSERT INTO TABELAFUNCIONARIO (TFCNCGFCA, TFCNCGTP, TFCNCOMS)
     VALUES (:FCANCODG, :TPNCODG, IIF((:ZERA = 'T') OR (:PRAZO < 48),0, :PERC_COMIS));
  END
  
  FOR SELECT DISTINCT CMSNCGTFC FROM COMISSAO_SINERGIA
  INTO :TIPO
  DO BEGIN
     INSERT INTO COMISSAO_SINERGIA (CMSNCGTFC, CMSNCGTP)
     VALUES (:TIPO, :TPNCODG);
  END


END^

SET TERM ; ^
