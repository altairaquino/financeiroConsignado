CREATE PROCEDURE SP_ADD_AJUDACUSTO(
  AJCNCODG INTEGER,
  AJCNCGEMP INTEGER,
  AJCNCGTAC INTEGER,
  AJCYVALOR FLOAT)
AS
BEGIN
IF(AJCNCODG IS NULL)THEN
  INSERT INTO
  AJUDACUSTO(
  AJCNCGEMP,
  AJCNCGTAC,
  AJCYVALOR)
VALUES(
  :AJCNCGEMP,
  :AJCNCGTAC,
  :AJCYVALOR);
ELSE
BEGIN
  IF(AJCYVALOR IS NOT NULL)THEN
  UPDATE
  AJUDACUSTO
  SET
  AJCYVALOR = :AJCYVALOR
  WHERE
  AJCNCODG = :AJCNCODG;
  ELSE
  DELETE
  FROM
  AJUDACUSTO
  WHERE
  AJCNCODG = :AJCNCODG;
END
END^
