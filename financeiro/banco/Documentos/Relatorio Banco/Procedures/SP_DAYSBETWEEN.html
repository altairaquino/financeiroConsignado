CREATE PROCEDURE SP_DAYSBETWEEN(
  DATA1 DATE,
  DATA2 DATE)
RETURNS(
  DAYS INTEGER)
AS
DECLARE VARIABLE AUX DATE;
BEGIN
 IF(DATA1 > DATA2)THEN
 BEGIN
      AUX = DATA1;
      DATA1 = DATA2;
      DATA2 = AUX;
 END
 DAYS = 0;
 AUX = DATA1;
 WHILE(AUX < DATA2)DO
 BEGIN
      DAYS = DAYS + 1;
      AUX = CAST(addDay(DATA1,DAYS) AS DATE);

 END
 SUSPEND;
END^

SET TERM ; ^
