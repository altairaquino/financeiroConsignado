CREATE PROCEDURE SP_ADD_FILTROAJCUST(
  FACNCODG INTEGER,
  FACNCGAJC INTEGER,
  FACNCGFIL INTEGER)
AS
BEGIN
IF(FACNCODG IS NULL)THEN
  INSERT INTO
  FILTROAJCUST(
  FACNCGAJC,
  FACNCGFIL)
VALUES(
  :FACNCGAJC,
  :FACNCGFIL);
ELSE
BEGIN 
  DELETE
  FROM
  FILTROAJCUST
  WHERE
  FACNCODG = :FACNCODG;
END
END^
