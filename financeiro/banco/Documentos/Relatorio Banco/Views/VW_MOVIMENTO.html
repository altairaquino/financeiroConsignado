CREATE VIEW VW_MOVIMENTO(
  MOVNCODG,
  MOVDDATA,
  MOVDVENC,
  MOVNCGGPC,
  MOVCDCGPC,
  MOVNCGCOG,
  MOVCDCCOG,
  MOVNCGORE,
  MOVCDCORE,
  MOVNFLQU,
  MOVCFLQU,
  MOVNQTPC,
  MOVNNRPC,
  MOVNFILI,
  MOVCFILI,
  MOVCDESC,
  MOVCDOCM,
  MOVYVALR,
  MOVNCADT,
  MOVCCADT,
  MOVDCADT,
  MOVNFORN,
  MOVCFORN,
  MOVNCOCO,
  MOVCCOCO,
  MOVLTIPO,
  MOVLATIV)
AS
SELECT
   MOVNCODG,
   MOVDDATA,
   MOVDVENC,
   GPCNCODG,
   GPCCDESC,
   MOVNCGCOG,
   COGCDESC,
   MOVNCGORE,
   MOVNFLQU,
   FLQCDESC,
   ORECDESC,
   MOVNQTPC,
   MOVNNRPC,
   MOVNFILI,
   FLCNOME,
   MOVCDESC,
   MOVCDOCM,
   MOVYVALR,
   MOVNCADT,
   CD.ENCNOME,
   MOVDCADT,
   MOVNFORN,
   FN.ENCNOME,
   MOVNCOCO,
   (BCCDESC ||' - AG:'|| COCAGEN ||' - C:'||COCNUMR||' ('||TCTCDESC||')') MOVCCOCO,
   MOVLTIPO,
   MOVLATIV
 FROM
   MOVIMENTO
   LEFT JOIN FORMALIQUIDEZ ON
     MOVNFLQU = FLQNCODG
   INNER JOIN CONTAGRUPO ON
      COGNCODG = MOVNCGCOG
   INNER JOIN GRUPOCONTA ON
      COGNCGGPC = GPCNCODG
   LEFT JOIN ORIGEMENTRADA ON
      ORENCODG = MOVNCGORE
   LEFT JOIN FILIAL ON
     FLNCODG = MOVNFILI
   LEFT JOIN ENTIDADE CD ON
     MOVNCADT = CD.ENNCODG
   LEFT JOIN ENTIDADE FN ON
     MOVNFORN = FN.ENNCODG
   LEFT JOIN CONTACORRENTE ON
     MOVNCOCO = CONCODG
   LEFT JOIN TIPOCONTACORRENTE ON
     CONTCON = TCTNCODG
   LEFT JOIN BANCO ON
     CONCGBC = BCNCODG
;
