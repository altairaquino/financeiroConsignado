CREATE VIEW VW_ANDAMENTOPROCESSO(
  ANPNCODG,
  ANPNCGPRO,
  ANPCNRPRO,
  ANPNCGFAP,
  ANPCDCFAP,
  ANPNCGEN,
  ANPCNMEN,
  ANPDDATA,
  ANPCDESC)
AS
SELECT
   ANPNCODG,
   ANPNCGPRO,
   PROCNUMR,
   ANPNCGFAP,
   FAPCDESC,
   ANPNCGEN,
   ENCNOME,
   ANPDDATA,
   ANPCDESC
 FROM
   ANDAMENTOPROCESSO INNER JOIN PROCESSO ON
      PRONCODG = ANPNCGPRO
   INNER JOIN FASEPROCESSO ON
      FAPNCODG = ANPNCGFAP
   INNER JOIN ENTIDADE ON
      ANPNCGEN = ENNCODG
 ORDER BY ANPDDATA DESC
;
