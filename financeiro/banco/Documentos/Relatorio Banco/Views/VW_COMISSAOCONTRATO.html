CREATE VIEW VW_COMISSAOCONTRATO(
  CCNCODG,
  CCNCGCT,
  CCNMRCT,
  CCYVLCT,
  CCNCGEN,
  CCCNMEN,
  CCCDOEN,
  CCCDCTN,
  CCDCADT,
  CCNPERC,
  CCNVLCM)
AS
SELECT
   CCNCODG,
   CCNCGCT,
   CTNNUMR,
   CTYVALR,
   CCNCGEN,
   ENCNOME,
   ENCDOCM,
   TNCDESC,
   CCDCADT,
   CCNPERC,
   (CTYVALR * CCNPERC / 100)CCNVLCM
 FROM
   COMISSAOCONTRATO LEFT JOIN CONTRATO ON
     CTNCODG = CCNCGCT
   LEFT JOIN ENTIDADE ON
     ENNCODG = CCNCGEN
   LEFT JOIN ANGARIADOR ON
     ANNCGEN = CCNCGEN
   LEFT JOIN TIPOANGARIADOR ON
     TNNCODG = ANNCGTN
;
