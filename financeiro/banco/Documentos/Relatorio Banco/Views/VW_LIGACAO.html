CREATE VIEW VW_LIGACAO(
  LGNCODG,
  LGNCGAN,
  LGCNMAN,
  LGNCGFC,
  LGCNMFC,
  LGNCGML,
  LGCDCML,
  LGNCGRL,
  LGCDCRL,
  LGCDOCM,
  LGDDATA,
  LGYVALR,
  LGMOBSV)
AS
SELECT
   LGNCODG,
   LGNCGEN,
   EN1.ENCNOME,
   LGNC2EN,
   EN2.ENCNOME,
   LGNCGML,
   MLCDESC,
   LGNCGRL,
   RLCDESC,
   LGCDOCM,
   LGDDATA,
   LGYVALR,
   LGMOBSV
 FROM
   LIGACAO LEFT JOIN ENTIDADE EN1 ON
     LGNCGEN = EN1.ENNCODG
   LEFT JOIN ENTIDADE EN2 ON
     LGNC2EN = EN2.ENNCODG
   LEFT JOIN MOTIVOLIGACAO ON
     LGNCGML = MLNCODG
   LEFT JOIN RESULTADOLIGACAO ON
     LGNCGRL = RLNCODG
;
