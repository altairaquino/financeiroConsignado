CREATE VIEW VW_USUARIOGRUPO(
  UGNCODG,
  UGNCGUS,
  UGCNMEN,
  UGNCGGR,
  UGCDCGR)
AS
SELECT
   UGNCODG,
   UGNCGUS,
   ENCNOME,
   UGNCGGR,
   GRCDESC
 FROM
   USUARIOGRUPO
   INNER JOIN ENTIDADE ON
      ENNCODG = UGNCGUS
   INNER JOIN GRUPO ON
      GRNCODG = UGNCGGR
;
