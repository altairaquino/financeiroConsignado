CREATE VIEW VW_FUNCREGIONAL(
  FRENCODG,
  FRENCGRG,
  FRECDCRG,
  FRENCGEN,
  FRECNMEN,
  FRENCGTFA,
  FRECDCTFA,
  FRELATIV)
AS
SELECT
   FRENCODG,
   RGNCODG,
   RGCDESC,
   FRENCGEN,
   ENCNOME,
   FRENCGTFA,
   TFACDESC,
   FRELATIV
 FROM
   FUNCREGIONAL
 INNER JOIN REGIONAL ON
      RGNCODG = FRENCGRG
 INNER JOIN ENTIDADE ON
      ENNCODG = FRENCGEN
 INNER JOIN TIPOFUNCAGENCIA ON
      TFANCODG = FRENCGTFA
 AND ENLATIV = 'T'
;
