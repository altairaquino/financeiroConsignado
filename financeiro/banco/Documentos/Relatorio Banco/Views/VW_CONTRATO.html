CREATE VIEW VW_CONTRATO(
  CTNCODG,
  CTNNUMR,
  CTNC2EN,
  CTCNMCL,
  CTCCPF,
  CTNCGEN,
  CTCNMAN,
  CTYVALR,
  CTNCOMS,
  CTYCOMS,
  CTDVERB,
  CTDPGTO,
  CTNCGFP,
  CTCDCFP,
  CTNCGSC,
  CTCDCSC,
  CTLPGCM,
  CTNCGPD,
  CTCDCPD,
  CTNCGTP,
  CTCDCTP,
  CTCMRTP,
  CTNQTPC,
  CTNEXAT,
  CTYEXAT,
  CTYSPRE,
  CTNPRAZ,
  CTLFISI,
  CTDBXFI,
  CTDEXTN,
  CTNCADT,
  CTCCADT,
  CTTCADT,
  CTDCADT,
  CTLAUDI,
  CTNCGBC,
  CTCDCBC,
  CTCAGEN,
  CTYTARI,
  CTDSQOP,
  CTNSQOP,
  CTCSQOP,
  CTLATIV)
AS
SELECT
   CTNCODG,
   CTNNUMR,
   CTNC2EN,
   EN1.ENCNOME,
   EN1.ENCDOCM,
   CTNCGEN,
   EN2.ENCNOME,
   CTYVALR,
   (SELECT SUM(COMISSAOCONTRATO.CCNPERC) FROM
     COMISSAOCONTRATO WHERE CCNCGCT = CTNCODG) CTNCOMS,
   (SELECT (SUM(COMISSAOCONTRATO.CCNPERC)*CTYVALR/100) FROM
     COMISSAOCONTRATO WHERE CCNCGCT = CTNCODG) CTYCOMS,
   CTDVERB,
   CTDPGTO,
   CTNCGFP,
   FPCDESC,
   CTNCGSC,
   SCCDESC,
   SCLPGCM,
   PDNCODG,
   PDCDESC,
   CTNCGTP,
   TPCNUMR ||' - '||TPCDESC,
   TPCNUMR,
   TPNQTPC,
   CTNEXAT,
   (CTNEXAT*CTYVALR/100) CTYEXAT,
   (CTNEXAT*CTYVALR/100)- (SELECT (SUM(COMISSAOCONTRATO.CCNPERC)*CTYVALR/100) FROM
     COMISSAOCONTRATO WHERE CCNCGCT = CTNCODG) CTYSPRE,
   TPNPRAZ,
   CTLFISI,
   CTDBXFI,
   CTDEXTN,
   CTNCADT,
   EN3.ENCNOME,
   CAST(CTDCADT AS TIME) CTDCADT,
   CAST(CTDCADT AS DATE) CTDCADT,   
   CTLAUDI,
   CTNCGBC,
   BCCDESC,
   CTCAGEN,
   IIF((SELECT SUM(CCNPERC) FROM
     COMISSAOCONTRATO
     WHERE CCNCGCT = CTNCODG
     AND CCNCGEN = CTNCGEN) > 0 AND CTNCGFP = 2,
     IIF((SELECT (SUM(CCNPERC)*CTYVALR/100) FROM
     COMISSAOCONTRATO WHERE CCNCGCT = CTNCODG
     AND CCNCGEN = CTNCGEN) BETWEEN 0 AND 100,2,4),0) CTYTARI,
   CTDSQOP,
   CTNSQOP,
   EN4.ENCNOME,
   CTLATIV
 FROM
   CONTRATO LEFT JOIN ENTIDADE EN1 ON
     CTNC2EN = EN1.ENNCODG
   LEFT JOIN ENTIDADE EN2 ON
     CTNCGEN = EN2.ENNCODG
   LEFT JOIN ENTIDADE EN3 ON
     CTNCADT = EN3.ENNCODG
   LEFT JOIN ENTIDADE EN4 ON
     CTNSQOP = EN4.ENNCODG
   LEFT JOIN FORMAPAGAMENTO ON
     FPNCODG = CTNCGFP
   LEFT JOIN SITUACAOCONTRATO ON
     SCNCODG = CTNCGSC
   LEFT JOIN TABELAPRODUTO ON
     TPNCODG = CTNCGTP
   LEFT JOIN PRODUTO ON
     TPNCGPD = PDNCODG
   LEFT JOIN BANCO ON
     BCNCODG = CTNCGBC
 --WHERE CTLATIV = 'T'
;
