CREATE VIEW VW_PRODUTO_ANGARIADOR(
  PDANCGEN,
  PDACNMEN,
  PDANCGPD,
  PDACDCPD)
AS
SELECT
   PDANCGEN,
   ENCNOME,
   PDANCGPD,
   PDCDESC
 FROM
   PRODUTO_ANGARIADOR LEFT JOIN PRODUTO ON
      PDANCGPD = PDNCODG
   LEFT JOIN ENTIDADE ON
      ENNCODG = PDANCGEN
;
