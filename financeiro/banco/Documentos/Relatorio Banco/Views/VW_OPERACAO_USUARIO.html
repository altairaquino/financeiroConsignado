CREATE VIEW VW_OPERACAO_USUARIO(
  OPNCODG,
  OPNPART,
  OPCDESC,
  OPCHINT,
  OPCLINK,
  OPNCGEN)
AS
SELECT
   OPNCODG,
   OPNPART,
   OPCDESC,
   OPCHINT,
   OPCLINK,
   ENNCODG
 FROM
   USUARIOOPERACAO LEFT JOIN OPERACAO ON
     OPNCODG = UONCGOP
   LEFT JOIN ENTIDADE ON
     ENNCODG = UONCGUS
 WHERE OPLATIV = 'T'
 UNION ALL
 SELECT
   OPNCODG,
   OPNPART,
   OPCDESC,
   OPCHINT,
   OPCLINK,
   ENNCODG
 FROM
   OPERACAOGRUPO LEFT JOIN OPERACAO ON
     OPNCODG = GONCGOP
   LEFT JOIN GRUPO ON
     GONCGGR = GRNCODG
   LEFT JOIN USUARIOGRUPO ON
     UGNCGGR = GRNCODG
   LEFT JOIN ENTIDADE ON
     ENNCODG = UGNCGUS
 WHERE OPLATIV = 'T'
;
