CREATE VIEW VW_PLANOCONTA(
  PLCNCODG,
  PLCNEMPR,
  PLCNPAI,
  PLCCPAI,
  PLCCDESC,
  PLCNTIPO,
  PLCLATIV)
AS
SELECT
   P2.PLCNCODG,
   P2.PLCNEMPR,
   P2.PLCNPAI,
   P1.PLCCDESC,
   P2.PLCCDESC,
   P2.PLCNTIPO,
   P2.PLCLATIV
 FROM
   PLANOCONTA P2 LEFT JOIN PLANOCONTA P1 ON
      P1.PLCNCODG = P2.PLCNPAI
