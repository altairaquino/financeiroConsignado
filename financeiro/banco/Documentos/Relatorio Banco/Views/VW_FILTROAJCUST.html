CREATE VIEW VW_FILTROAJCUST(
  FACNCGFIL,
  FACNCGAJC,
  FACNCGEN,
  FACCCDEN,
  FACLATV,
  FACNCODG,
  FACYVLAJC)
AS
SELECT
   FILNCODG FACNCGFIL,
   AJCNCODG FACNCGAJC,
   ENNCODG FACNCGEN,
   ENCNOME FACCCDEN,
   IIF(FACNCGAJC IS NULL, 'F', 'T') FACLATV,
   FACNCODG,
   AJCYVALOR
 FROM
   AJUDACUSTO INNER JOIN EMPREGADO on
   EMPNCODG = AJCNCGEMP
   INNER JOIN ENTIDADE ON
   ENNCODG = EMPNCGCEN
   INNER JOIN FILTRO ON
   FILNCODG IS NOT NULL
   LEFT JOIN FILTROAJCUST ON
   FACNCGAJC = AJCNCODG AND
   FACNCGFIL = FILNCODG
 WHERE
   AJCLATIV = 'T'
;
