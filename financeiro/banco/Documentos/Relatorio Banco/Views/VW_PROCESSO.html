CREATE VIEW VW_PROCESSO(
  PRONCODG,
  PRONCODG2,
  PRONCGEN,
  PROCNMCL,
  PRODNASC,
  PROCCPF,
  PROCSEXO,
  PRONC2EN,
  PROCNMCN,
  PROCNUMR,
  PRONCGTAC,
  PROCDCTAC,
  PROCLOCA,
  PROCOBS,
  PRONCADT,
  PROCCADT,
  PRONALTE,
  PROCALTE,
  PRODDATA,
  PRODALTE,
  PROCUFCD,
  PROCDCCD,
  PRONCGCD,
  PRONCGTL,
  PROCDCTL,
  PROCLOGR,
  PROCFONE,
  PROCFONE2,
  PROCCEP,
  PROCBAIR,
  PRONCONS,
  PROCCONS,
  PRONSUPE,
  PROCSUPE,
  PROLATIV)
AS
SELECT
   PRONCODG,
   PRONCODG2,
   PRONCGEN,
   CL.ENCNOME,
   CL.ENDNASC,
   CL.ENCDOCM,
   CL.ENCSEXO,
   PRONC2EN,
   CN.ENCNOME,
   PROCNUMR,
   PRONCGTAC,
   TACCDESC,
   PROCLOCA,
   PROCOBS,
   PRONCADT,
   US.ENCNOME,
   PRONALTE,
   EN.ENCNOME,
   PRODDATA,
   PRODALTE,
   CDCUF,
   CDCDESC,
   CDNCODG,
   TLNCODG,
   TLCDESC,
   CL.ENLENDR,
   CL.ENCFONE,
   CL.ENCCELL,
   CL.ENCCEP,
   CL.ENCBAIR,
   PRONCONS,
   CS.ENCNOME,
   PRONSUPE,
   SU.ENCNOME,
   PROLATIV
 FROM
   PROCESSO INNER JOIN ENTIDADE CL ON
      CL.ENNCODG = PRONCGEN
   LEFT JOIN ENTIDADE CN ON
      CN.ENNCODG = PRONC2EN
   INNER JOIN TIPOACAOJUDICIAL ON
      TACNCODG = PRONCGTAC
   INNER JOIN ENTIDADE US ON
      US.ENNCODG = PRONCADT
   LEFT JOIN ENTIDADE EN ON
      EN.ENNCODG = PRONALTE
   LEFT JOIN CIDADE ON
      CDNCODG = CL.ENNCGCD
   LEFT JOIN TIPOLOGRADOURO ON
      TLNCODG = CL.ENNCGTL
   LEFT JOIN ENTIDADE CS ON
      CS.ENNCODG = PRONCONS
   LEFT JOIN ENTIDADE SU ON
      SU.ENNCODG = PRONSUPE
;
