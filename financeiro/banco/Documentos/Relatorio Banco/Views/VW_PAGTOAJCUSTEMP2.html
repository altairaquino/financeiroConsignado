CREATE VIEW VW_PAGTOAJCUSTEMP2(
  PCENCODG,
  PCENCGEMP,
  PCENCGTAC,
  PCENCGPAC,
  PCEYVALOR,
  PCEDPAGTO,
  PCEYACRES,
  PCEYDESC,
  PCEYTOTL,
  PCENCGEN,
  PCECCDEN,
  PCECDCTAC,
  PCELBLOQ,
  PCENCGCO,
  PCENCGCTC,
  PCENCGBC,
  PCECAGEN,
  PCECNUM,
  PCECTITU,
  PCECDCTCT,
  PCECCDBC)
AS
SELECT
	PCENCODG,
	PCENCGEMP,
	PCENCGTAC,
	PCENCGPAC,
	PCEYVALOR,
	PCEDPAGTO,
	PCEYACRES,
	PCEYDESC,
	PCEYTOTL,
	ENNCODG PCENCGEN,
	ENCNOME PCECCDEN,
	TACCDESC PCECDCTAC,
	PCELBLOQ,
    CONCODG PCENCGCO,
    CONTCON PCENCGCTC,
    CONCGBC PCENCGBC,
    COCAGEN PCECAGEN,
    COCNUMR PCECNUM,
    COCTITU PCECTITU,
    TCTCDESC PCECDCTCT,
    BCCDESC PCECCDBC
 FROM PAGTOAJCUSTEMP LEFT JOIN EMPREGADO ON
   PCENCGEMP = EMPNCODG
   LEFT JOIN ENTIDADE ON
   EMPNCGCEN = ENNCODG
   LEFT JOIN TIPOAJCUST ON
   PCENCGTAC = TACNCODG
   LEFT JOIN CONTACORRENTE ON
   CONCGEN = ENNCODG AND
   COLPRIN = 'T'
   LEFT JOIN TIPOCONTACORRENTE ON
   CONTCON = TCTNCODG
   LEFT JOIN BANCO ON
   BCNCODG = CONCGBC
 WHERE
   PCELBLOQ = 'F'
;
