CREATE VIEW VW_TABELAPRODUTO(
  TPNCODG,
  TPNCGPD,
  TPCDCPD,
  TPNCGBC,
  TPCABPD,
  TPCNUMR,
  TPCDESC,
  TPCDESC2,
  TPNQTPC,
  TPNCARE,
  TPNPRAZ,
  TPNIMPT,
  TPLZERA,
  TPNEXAT,
  TPNCATB,
  TPNLIQU)
AS
SELECT
   TPNCODG,
   TPNCGPD,
   PDCDESC,
   PDNCGBC,
   PDCABRV,
   TPCNUMR,
   TPCDESC,
   TPCDESC||' - ('||TPNQTPC||' X)' TPCDESC,
   TPNQTPC,
   TPNCARE,
   TPNPRAZ,
   TPNIMPT,
   TPLZERA,
   TPNEXAT,
   (TPNEXAT * TPNIMPT / 100) TPNCATB,
   (TPNEXAT - (TPNEXAT * TPNIMPT / 100)) TPNLIQU
 FROM
   TABELAPRODUTO LEFT JOIN PRODUTO ON
     PDNCODG = TPNCGPD
 WHERE
   TPLATIV = 'T'
;
