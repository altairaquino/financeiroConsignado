CREATE VIEW VW_ATIVIDADE(
  ATINCODG,
  ATINCGEN,
  ATICNMEN,
  ATIDDATA,
  ATITHORA,
  ATICLOCA,
  ATICCONT,
  ATICFONE,
  ATICDESC,
  ATICOBSV,
  ATICPERS,
  ATIYGAST,
  ATINODIN,
  ATINODFN,
  ATINKMRD,
  ATINCANC)
AS
SELECT
   ATINCODG,
   ATINCGEN,
   ENCNOME,
   ATIDDATA,
   ATITHORA,
   ATICLOCA,
   ATICCONT,
   ATICFONE,
   ATICDESC,
   ATICOBSV,
   ATICPERS,
   ATIYGAST,
   ATINODIN,
   ATINODFN,
   CAST((ATINODFN - ATINODIN)as float) ATINKMRD,
   ATINCANC
 FROM
   ATIVIDADE INNER JOIN ENTIDADE ON
      ATINCGEN = ENNCODG
;
