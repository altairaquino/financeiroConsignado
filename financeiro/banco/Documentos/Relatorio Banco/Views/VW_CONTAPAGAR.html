CREATE VIEW VW_CONTAPAGAR(
  CPNCODG,
  CPDDATA,
  CPNFLQU,
  CPCFLQU,
  CPNCONT,
  CPCCONT,
  CPNFILI,
  CPCFILI,
  CPNFORN,
  CPCFORN,
  CPNCOCO,
  CPCCOCO,
  CPCDESC,
  CPCDOCM,
  CPYVALR,
  CPDPGTO,
  CPNCGEN,
  CPCNMEN,
  CPLATIV)
AS
SELECT
   CPNCODG,
   CPDDATA,
   CPNFLQU,
   FLQCDESC,
   CPNCONT,
   PLCCDESC,
   FLNCODG,
   FLCNOME,
   CPNFORN,
   E1.ENCNOME,
   CONCODG,
   (BCCDESC ||' - AG:'|| COCAGEN ||' - C:'||COCNUMR||' ('||TCTCDESC||')') COCDESC,
   CPCDESC,
   CPCDOCM,
   CPYVALR,
   CPDPGTO,
   CPNCGEN,
   E2.ENCNOME,
   CPLATIV
 FROM
   CONTAPAGAR LEFT JOIN FORMALIQUIDEZ ON
     CPNFLQU = FLQNCODG
   LEFT JOIN PLANOCONTA ON
     PLCNCODG = CPNCONT
   LEFT JOIN ENTIDADE E1 ON
     E1.ENNCODG = CPNFORN
   LEFT JOIN FILIAL ON
     CPNFILI = FLNCODG
   LEFT JOIN ENTIDADE E2 ON
     E2.ENNCODG = CPNCGEN
   LEFT JOIN CONTACORRENTE ON
     CPNCOCO = CONCODG
   LEFT JOIN BANCO ON
     CONCGBC = BCNCODG
   LEFT JOIN TIPOCONTACORRENTE ON
     TCTNCODG = CONTCON
 WHERE CPLATIV = 'T'
;
