CREATE VIEW VW_PAGTOAJCUSTEMP(
  PCENCODG,
  PCENCGEMP,
  PCENCGTAC,
  PCENCGPAC,
  PCEYVALOR,
  PCEDPAGTO,
  PCEYACRES,
  PCEYDESC,
  PCENCGEN,
  PCECCDEN,
  PCECDCTAC,
  PCELBLOQ)
AS
SELECT
	PCENCODG,
	PCENCGEMP,
	PCENCGTAC,
	PCENCGPAC,
	PCEYVALOR,
	PCEDPAGTO,
	PCEYACRES,
	PCEYDESC,

PCENCGTAC = TACNCODG
