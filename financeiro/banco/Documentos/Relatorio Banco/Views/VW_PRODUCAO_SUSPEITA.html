CREATE VIEW VW_PRODUCAO_SUSPEITA(
  COD_AGENTE,
  NOME_AGENTE,
  META,
  PROD_DIA,
  QUANTID,
  META_DIA,
  PERC_30,
  SUSPEITO)
AS
SELECT
  ANNCGEN COD_AGENTE,
  ENCNOME NOME_AGENTE,
  ANNMETA META_AGENTE,
  sum(ctyvalr) PROD_DIA,
  COUNT(CTNCODG) QUANT,
  (ANNMETA/ 30) META_DIA,
  (ANNMETA * 30 / 100) PERC_30,
  IIF(sum(ctyvalr) > (ANNMETA * 30 / 100), 'T','F') SUSPEITO
FROM CONTRATO INNER JOIN ENTIDADE ON
     ENNCODG = CTNCGEN
   INNER JOIN ANGARIADOR ON
     ENNCODG = ANNCGEN
WHERE CAST(CTDCADT AS DATE) = 'NOW'
AND ANNCGTN = 3
group by ANNCGEN, ENCNOME, ANNMETA
;
