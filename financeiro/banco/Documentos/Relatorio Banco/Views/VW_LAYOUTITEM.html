CREATE VIEW VW_LAYOUTITEM(
  LITNIDLAY,
  LITNBEGN,
  LITNEND,
  LITCVALD,
  LITCFORM,
  LITCNMATR,
  LITNIDCLA,
  LITCNMCLA,
  LITLINIT,
  LITNIDTIP)
AS
SELECT
   LITNIDLAY,
   LITNBEGN,
   LITNEND,
   LITCVALD,
   LITCFORM,
   ATRCNAME,
   ATRNIDCLA,
   CLACNAME,
   LITLINIT,
   LITNIDTIP
 FROM
   LAYOUTITEM
   LEFT JOIN  ATRIBUTTE ON
   LITNIDATR = ATRNID
   LEFT JOIN CLASS ON
   ATRNTIPO = CLANID
   LEFT JOIN TIPO ON
   TIPNID = LITNIDTIP
;
