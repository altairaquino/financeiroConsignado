CREATE VIEW VW_CONTAFIXA(
  COFNCODG,
  COFNCGGPC,
  COFCDCGPC,
  COFNCONT,
  COFCCONT,
  COFNFILI,
  COFCFILI,
  COFNDIA,
  COFCDESC,
  COFCDOCM,
  COFNFORN,
  COFCFORN,
  COFNCOCO,
  COFCCOCO,
  COFYVALR,
  COFLATIV)
AS
SELECT
   COFNCODG,
   GPCNCODG,
   GPCCDESC,
   COFNCONT,
   COGCDESC,
   COFNFILI,
   FLCNOME,
   COFNDIA,
   COFCDESC,
   COFCDOCM,
   COFNFORN,
   ENCNOME,
   COFNCOCO,
   (BCCDESC ||' - AG:'|| COCAGEN ||' - C:'||COCNUMR||' ('||TCTCDESC||')') COFCCOCO,
   COFYVALR,
   COFLATIV
 FROM
   CONTAFIXA INNER JOIN CONTAGRUPO ON
      COGNCODG = COFNCONT
   INNER JOIN GRUPOCONTA ON
      COGNCGGPC = GPCNCODG
   LEFT JOIN FILIAL ON
      FLNCODG = COFNFILI
   LEFT JOIN ENTIDADE ON
      COFNFORN = ENNCODG
   LEFT JOIN CONTACORRENTE ON
      COFNCOCO = CONCODG
   LEFT JOIN TIPOCONTACORRENTE ON
      CONTCON = TCTNCODG
   LEFT JOIN BANCO ON
      CONCGBC = BCNCODG
 WHERE COFLATIV = 'T'
;
