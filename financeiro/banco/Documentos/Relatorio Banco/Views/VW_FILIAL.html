CREATE VIEW VW_FILIAL(
  FLNCODG,
  FLNCGEP,
  FLCNMEP,
  FLCNOME,
  FLCRESP,
  FLCFONE,
  FLCMAIL,
  FLNCGTL,
  FLCDCTL,
  FLCLOGR,
  FLCBAIR,
  FLNCGCD,
  FLCDCCD,
  FLCUFCD,
  FLCCEP)
AS
SELECT
   FLNCODG,
   FLNCGEP,
   EPCNOME,
   FLCNOME,
   FLCRESP,
   FLCFONE,
   FLCMAIL,
   FLNCGTL,
   TLCDESC,
   FLCLOGR,
   FLCBAIR,
   FLNCGCD,
   CDCDESC,
   CDCUF,
   FLCCEP
 FROM
   FILIAL INNER JOIN EMPRESA ON
      EPNCODG = FLNCGEP
   INNER JOIN CIDADE ON
      FLNCGCD = CDNCODG
   LEFT JOIN TIPOLOGRADOURO ON
      FLNCGTL = TLNCODG
;
