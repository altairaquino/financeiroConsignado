CREATE VIEW VW_CENTROCONTAPAGAR(
  CRCNCODG,
  CRCNCGCP,
  CRCCCONT,
  CRCNCGCR,
  CRCCCENT,
  CRCYVALR,
  CRCDDATA)
AS
SELECT
   CRCNCODG,
   CRCNCGCP,
   PLCCDESC,
   CRCNCGCR,
   CRCDESC,
   CPYVALR,
   CRCDDATA
 FROM
   CENTROCONTAPAGAR LEFT JOIN CONTAPAGAR ON
     CPNCODG = CRCNCGCP
   LEFT JOIN CENTROCUSTO ON
     CRNCODG = CRCNCGCR
   LEFT JOIN PLANOCONTA ON
     PLCNCODG = CPNCONT
