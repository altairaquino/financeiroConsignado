CREATE VIEW VW_CONTRATOAUTO(
  CTANCODG,
  CTACNUMR,
  CTANCGEN,
  CTACNMEN,
  CTACFONE,
  CTACCELL,
  CTACDOCM,
  CTANVEIC,
  CTACMAVE,
  CTACMODE,
  CTANCGEP,
  CTACDCEP,
  CTANCGOP,
  CTACNMOP,
  CTANCGLJ,
  CTACNMLJ,
  CTAYVALR,
  CTANPLAN,
  CTAYVLPC,
  CTANFPTO,
  CTACFPTO,
  CTADBASE,
  CTACOBSV,
  CTADCADT,
  CTALCARN,
  CTALATEN,
  CTALVEIC,
  CTALINDI,
  CTACOBIN,
  CTALPDCN,
  CTALPDPC,
  CTALATIV)
AS
SELECT
   CTANCODG,
   CTACNUMR,
   CTANCGEN,
   EN1.ENCNOME,
   EN1.ENCFONE,
   EN1.ENCCELL,
   EN1.ENCDOCM,
   CTANVEIC,
   MVECDESC,
   VEICMODE,
   CTANCGEP,
   EPCNOME,
   CTANC2EN,
   EN2.ENCNOME,
   CTANC3EN,
   EN3.ENCNOME,
   CTAYVALR,
   CTANPLAN,
   CTAYVLPC,
   CTANFPTO,
   PGTCDESC,
   CTADBASE,
   CTACOBSV,
   CTADCADT,
   CTALCARN,
   CTALATEN,
   CTALVEIC,
   CTALINDI,
   CTACOBIN,
   IIF (CTALCARN = 'F' AND CTADCARN < 'NOW','T','F') CTALPDCN,
   IIF (EXISTS(SELECT PCCNCODG FROM VW_PARCCONTRATOAUTO
               WHERE PCCNCTAU = CTANCODG
               AND (IIF(PCCDPRAZ = PCCDVENC, ADDDAY(PCCDPRAZ,10), CAST(PCCDPRAZ AS TIMESTAMP)) < 'NOW'
               AND ADDDAY(PCCDVENC,10) < 'NOW')
               AND PCCDPGTO IS NULL),'T','F') CTALPDPC,
   CTALATIV
 FROM
   CONTRATOAUTO LEFT JOIN ENTIDADE EN1 ON
      CTANCGEN = EN1.ENNCODG
   LEFT JOIN ENTIDADE EN2 ON
      CTANC2EN = EN2.ENNCODG
   LEFT JOIN ENTIDADE EN3 ON
      CTANC3EN = EN3.ENNCODG
   LEFT JOIN PGTOCONTRATOAUTO ON
      CTANFPTO = PGTNCODG
   LEFT JOIN VEICULO ON
      VEINCODG = CTANVEIC
   LEFT JOIN MARCAVEICULO ON
      VEINMARC = MVENCODG
   LEFT JOIN EMPRESA ON
      EPNCODG = CTANCGEP
;
