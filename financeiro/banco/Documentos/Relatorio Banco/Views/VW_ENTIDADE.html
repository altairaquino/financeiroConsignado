CREATE VIEW VW_ENTIDADE(
  ENNCODG,
  ENCNOME,
  ENNCGTE,
  ENCDCTE,
  ENCSGTE,
  ENNCGPP,
  ENCDCPP,
  ENCTPDC,
  ENCSEXO,
  ENCLOGN,
  ENCSENH,
  ENCDOCM,
  ENDEXRG,
  ENCRG,
  ENCOERG,
  ENCUFRG,
  ENCDORG,
  ENDNASC,
  ENCCONJ,
  ENCAPEL,
  ENNCGEC,
  ECCDESC,
  ENCPAI,
  ENCMAE,
  ENNCGTL,
  ENCDCTL,
  ENLENDR,
  ENCFONE,
  ENCCELL,
  ENCMAIL,
  ENNCGCD,
  ENCUFCD,
  ENCDCCD,
  ENCCEP,
  ENCBAIR,
  ENMOBS,
  ENLATIV)
AS
SELECT
   ENNCODG,
   ENCNOME,
   ETNCGTE,
   TECDESC,
   TECSIGL,
   ENNCGPP,
   PPCDESC,
   PPCTPDC,
   ENCSEXO,
   ENCLOGN,
   ENCSENH,
   ENCDOCM,
   ENDEXRG,
ENCRG  ,
ENCOERG,
ENCUFRG,
ENCDORG,
   ENDNASC,
   ENCCONJ,
   ENCAPEL,
   ENNCGEC,
   ECCDESC,
   ENCPAI,
   ENCMAE,
   ENNCGTL,
   TLCDESC,
   ENLENDR,
   ENCFONE,
   ENCCELL,
   ENCMAIL,
   ENNCGCD,
   CDCUF,
   CDCDESC,
   ENCCEP,
   ENCBAIR,
   ENMOBS,
   ENLATIV
 FROM
   ENTIDADETIPO LEFT JOIN ENTIDADE ON
     ETNCGEN = ENNCODG
   LEFT JOIN TIPOENTIDADE ON
     ETNCGTE = TENCODG
   LEFT JOIN TIPOPESSOA ON
     ENNCGPP = PPNCODG
   LEFT JOIN ESTADOCIVIL ON
     ENNCGEC = ECNCODG
   LEFT JOIN TIPOLOGRADOURO ON
     ENNCGTL = TLNCODG
   LEFT JOIN CIDADE ON
     ENNCGCD = CDNCODG
 WHERE
   ENLATIV = 'T'
;
