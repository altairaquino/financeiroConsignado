CREATE VIEW VW_TABELAFUNCIONARIO(
  TFCNCODG,
  TFCNCGFCA,
  TFCNCGFC,
  TFCCNMFC,
  TFCNCGAG,
  TFCCNMAG,
  TFCNCGPD,
  TFCCDCPD,
  TFCCABPD,
  TFCNCGTP,
  TFCCNMTP,
  TFCCDCTP,
  TFCNPRAZ,
  TFCNCARE,
  TFCNCOMS,
  TFCNCM70,
  TFCNCM100)
AS
SELECT
   TFCNCODG,
   FCANCODG,
   E1.ENNCODG,
   E1.ENCNOME,
   E2.ENNCODG,
   E2.ENCNOME,
   PDNCODG,
   PDCDESC,
   PDCABRV,
   TPNCODG,
   TPCNUMR,
   TPCDESC,
   TPNPRAZ,
   TPNCARE,
   TFCNCOMS,
   TFCNCM70,
   TFCNCM100
 FROM
   TABELAFUNCIONARIO LEFT JOIN FUNCIONARIOAGENCIA ON
     TFCNCGFCA = FCANCODG
   LEFT JOIN ENTIDADE E1 ON
     FCANC2EN = E1.ENNCODG
   LEFT JOIN ENTIDADE E2 ON
     FCANCGEN = E2.ENNCODG
   LEFT JOIN TABELAPRODUTO ON
     TPNCODG = TFCNCGTP
   LEFT JOIN PRODUTO ON
     PDNCODG = TPNCGPD
;
