CREATE VIEW VW_MOVCAIXA(
  MOVNCODG,
  MOVNEMPR,
  MOVDDATA,
  MOVDVENC,
  MOVNFLQU,
  MOVCFLQU,
  MOVNCONT,
  MOVCCONT,
  MOVNCGFL,
  MOVCDCFL,
  MOVNTIPO,
  MOVCDESC,
  MOVCDOCM,
  MOVYVALR,
  MOVDCONS,
  MOVNCGEN,
  MOVCNMEN,
  MOVNFORN,
  MOVCFORN,
  MOVNCOCO,
  MOVCCOCO,
  MOVLATIV)
AS
SELECT
   MOVNCODG,
   MOVNEMPR,
   MOVDDATA,
   MOVDVENC,
   MOVNFLQU,
   FLQCDESC,
   MOVNCONT,
   PLCCDESC,
   MOVNFILI,
   FLCNOME,
   PLCNTIPO,
   MOVCDESC,
   MOVCDOCM,
   MOVYVALR,
   MOVDCONS,
   MOVNCGEN,
   E1.ENCNOME,
   MOVNFORN,
   E2.ENCNOME,
   MOVNCOCO,
   (BCCDESC ||' - AG:'|| COCAGEN ||' - C:'||COCNUMR||' ('||TCTCDESC||')') MOVCCOCO,
   MOVLATIV
 FROM
   MOVCAIXA LEFT JOIN FORMALIQUIDEZ ON
     MOVNFLQU = FLQNCODG
   LEFT JOIN PLANOCONTA ON
     PLCNCODG = MOVNCONT
   LEFT JOIN FILIAL ON
     FLNCODG = MOVNFILI
   LEFT JOIN ENTIDADE E1 ON
     MOVNCGEN = E1.ENNCODG
   LEFT JOIN ENTIDADE E2 ON
     MOVNFORN = E2.ENNCODG
   LEFT JOIN CONTACORRENTE ON
     MOVNCOCO = CONCODG
   LEFT JOIN TIPOCONTACORRENTE ON
     CONTCON = TCTNCODG
   LEFT JOIN BANCO ON
     CONCGBC = BCNCODG
WHERE
MOVLATIV = 'T'
;
