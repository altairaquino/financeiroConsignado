CREATE VIEW VW_TABELA_ANGARIADOR(
  TGNCODG,
  TGNCGEN,
  TGCNMEN,
  TGNCGPD,
  TGCDCPD,
  TGCABPD,
  TGNCGTP,
  TGCNMTP,
  TGCDCTP,
  TGNPRAZ,
  TGNCARE,
  TGNLIQU,
  TGNCOMS)
AS
SELECT
   TGNCODG,
   TGNCGEN,
   ENCNOME,
   TPNCGPD,
   PDCDESC,
   PDCABRV,
   TGNCGTP,
   TPCNUMR,
   TPCDESC,
   TPNPRAZ,
   TPNCARE,
   (TPNEXAT - (TPNEXAT * TPNIMPT / 100)) TPNLIQU,
   TGNCOMS
 FROM
   TABELA_ANGARIADOR LEFT JOIN ENTIDADE ON
     ENNCODG = TGNCGEN
   LEFT JOIN TABELAPRODUTO ON
     TPNCODG = TGNCGTP
   LEFT JOIN PRODUTO ON
     PDNCODG = TPNCGPD
;
